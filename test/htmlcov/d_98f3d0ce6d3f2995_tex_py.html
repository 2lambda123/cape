1""" 

2:mod:`cape.filecntl.tex`: Interface to LaTeX files 

3===================================================== 

4 

5This is a module built off of the :class:`cape.filecntl.FileCntl` class 

6customized for manipulating Cape's automated reports. In addition to 

7containing methods for reading, writing, and organizing lines of text, 

8it contains a :func:`Compile` method for compiling the PDF and sparing 

9the user from constructing the system command to do so. 

10 

11See the :mod:`cape.cfdx.report` module or the 

12:ref:`JSON report page <cape-json-Report>` page for more information. 

13 

14""" 

15 

16# Standard library 

17import os 

18import subprocess as sp 

19 

20# Third-party modules 

21import numpy as np

23# Local modules

26# Base this class off of the main file control class. 

27class Tex(FileCntl): 

28 """ 

29 File control class for LaTeX files 

30  

31 :Call: 

32 >>> TX = cape.tex.Tex() 

33 >>> TX = cape.tex.Tex(fname='report.tex') 

34 :Inputs: 

35 *fname*: :class:`str` 

36 Name of LaTex report file to read/write 

37 :Outputs: 

38 *TX*: :class:`cape.tex.Tex` 

39 Instance of LaTeX report interface 

40 :Versions: 

41 * 2015-03-07 ``@ddalle``: Started 

42 * 2015-03-10 ``@ddalle``: First version 

43 """ 

44 

45 # Initialization method (not based off of FileCntl) 

46 def __init__(self, fname='report.tex'): 

47 """Initialization method""" 

48 # Read the file. 

49 self.Read(fname) 

50 # Save the file name. 

51 self.fname = fname 

52 # Save the location 

53 self.fdir = os.path.split(os.path.abspath(fname))[0] 

54 # Split into sections. 

55 self.SplitToSections(reg="%\$__([\w_]+)") 

56 

57 # Display method 

58 def __repr__(self): 

59 """Display method for LaTeX class 

60  

61 :Versions: 

62 * 2015-03-07 ``@ddalle``: First version 

63 """ 

64 # Initialize the string. 

65 s = '<Tex("%s", %i lines' % (self.fname, len(self.lines)) 

66 # Check for number of sections. 

67 if hasattr(self, 'SectionNames'): 

68 # Write the number of sections. 

69 s = s + ", %i sections)>" % len(self.SectionNames) 

70 else: 

71 # Just close the string. 

72 s = s + ")>" 

73 return s 

74 

75 # Function to compile. 

76 def Compile(self, check=True): 

77 """Compile the LaTeX file 

78  

79 :Call: 

80 >>> TX.Compile(check=True) 

81 :Inputs: 

82 *TX*: :class:`cape.tex.Tex` 

83 Instance of LaTeX report interface 

84 *check*: {``True``} | ``False`` 

85 Whether or not to fail if there is a nonzero compile status 

86 :Versions: 

87 * 2015-03-07 ``@ddalle``: First version 

88 * 2017-03-07 ``@ddalle``: Added *check* 

89 """ 

90 # Save current location. 

91 fpwd = os.getcwd() 

92 # Go to the file's location 

93 os.chdir(self.fdir) 

94 # Hide the output. 

95 f = open('/dev/null', 'w') 

96 # Compile. 

97 ierr = sp.call(['pdflatex', '-interaction=nonstopmode', self.fname], 

98 stdout=f) 

99 # Close the output "file". 

100 f.close() 

101 # Go back to original location 

102 os.chdir(fpwd) 

103 # Check compile status. 

104 if ierr and check: 

105 # Create an error 

106 raise SystemError("Compiling '%s' failed with status %i." 

107 % (self.fname, ierr)) 

108 elif ierr: 

109 # Throw a warning 

110 print("Compiling '%s' failed with status %i." 

111 % (self.fname, ierr))
