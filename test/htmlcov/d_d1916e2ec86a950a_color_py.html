176# Convert Hex to RGB 

177def Hex2RGB(col): 

178 r"""Convert hex-code color to RGB values 

179 

180 :Call: 

181 >>> rgb = Hex2RGB(col) 

182 >>> r, g, b = Hex2RGB(col) 

183 :Inputs: 

184 *col*: :class:`str` 

185 Six-digit hex code including ``'#'`` character 

186 :Outputs: 

187 *rgb*: :class:`list`\ [:class:`int`] 

188 * *len*: 3 

189 

190 Vector of RGB values 

191 

192 *r*: 0 <= :class:`int` <= 255 

193 Red value 

194 *g*: 0 <= :class:`int` <= 255 

195 Green value 

196 *b*: 0 <= :class:`int` <= 255 

197 Blue value 

198 :Versions: 

199 * 2017-01-05 ``@ddalle``: Version 1.0 

200 """ 

201 # Check first character 

202 if col[0] != '#': 

203 raise ValueError( 

204 ("Color '%s' is not a valid hex code\n" % col) + 

205 "Valid hex codes start with '#'") 

206 # Conversions 

207 r = eval('0x' + col[1:3]) 

208 g = eval('0x' + col[3:5]) 

209 b = eval('0x' + col[5:7]) 

210 # Output 

211 return [r, g, b]

214# Convert RGB to Hex code 

215def RGB2Hex(col): 

216 r"""Convert RGB integer values to a hex code 

217 

218 :Call: 

219 >>> hx = RGB2Hex(col) 

220 >>> hx = RGB2Hex((r,g,b)) 

221 :Inputs: 

222 *col*: :class:`tuple` | :class:`list` 

223 Indexable RGB specification 

224 *r*: 0 <= :class:`int` <= 255 

225 Red value 

226 *g*: 0 <= :class:`int` <= 255 

227 Green value 

228 *b*: 0 <= :class:`int` <= 255 

229 Blue value 

230 :Outputs: 

231 *hx*: :class:`str` 

232 Six-digit hex code including ``'#'`` character 

233 :Versions: 

234 * 2017-01-05 ``@ddalle``: Version 1.0 

235 """ 

236 # Create hex code 

237 hx = "#%02x%02x%02x" % (col[0], col[1], col[2]) 

238 # Output 

239 return hx

242# Convert generic type to RGB 

243def ToRGB(col): 

244 r"""Convert generic value to RGB 

245 

246 :Call: 

247 >>> rgb = ToRGB(col) 

248 >>> rgb = ToRGB(hx) 

249 >>> rgb = ToRGB(cname) 

250 >>> rgb = ToRGB(rgb) 

251 >>> rgb = ToRGB(RGB) 

252 :Inputs: 

253 *col*: :class:`str` | :class:`list` 

254 Generic color specification 

255 *hx*: :class:`str` 

256 Six-digit hex code starting with ``'#'`` character 

257 *rgb*: :class:`list`\ [:class:`int`] 

258 Vector of RGB values 

259 *cname*: :class:`str` 

260 Color by name, such as ``"orange"`` 

261 *V*: 0 <= :class:`float` <= 1 

262 Grayscale value 

263 *RGB*: :class:`list`\ [:class:`float`] 

264 Vector of fractional RGB values, 0 to 1 (inclusive) 

265 :Outputs: 

266 *rgb*: :class:`list`\ [:class:`int`] 

267 Vector of RGB values 

268 :Versions: 

269 * 2017-01-05 ``@ddalle``: Version 1.0 

270 """ 

271 # Get input type 

272 t = type(col).__name__ 

273 # Check input type 

274 if t in ['unicode', 'str']: 

275 # Convert to lower case and remove spaces 

276 col = col.lower() 

277 col = col.replace(' ', '') 

278 # Check for hex code 

279 if col.startswith('#'): 

280 # Convert from hex 

281 return Hex2RGB(col) 

282 elif col not in cnames: 

283 # Could not find 

284 raise ValueError( 

285 "Could not find a definition for color named '%s'" % col) 

286 else: 

287 # Get hex code from the dictionary 

288 return Hex2RGB(cnames[col]) 

289 elif t == ["float"]: 

290 # Grayscale 

291 return max(0, min(255, [int(255*col)] * 3)) 

292 elif t not in ['list', 'ndarray', 'tuple']: 

293 # Not an array 

294 raise TypeError( 

295 "Could not interpret a color that's not a string or list") 

296 elif len(col) != 3: 

297 # Not long enough 

298 raise ValueError("Color does not have three components") 

299 # Already RGB if reached this point; check for int 

300 r = col[0]; tr = type(r).__name__ 

301 g = col[1]; tg = type(g).__name__ 

302 b = col[2]; tb = type(b).__name__ 

303 # Check type 

304 if tr.startswith('float'): r = int(255*r) 

305 if tg.startswith('float'): g = int(255*g) 

306 if tb.startswith('float'): b = int(255*b) 

307 # Limit 

308 r = max(0, min(255, r)) 

309 g = max(0, min(255, g)) 

310 b = max(0, min(255, b)) 

311 # Output 

312 return [r, g, b]
